## Quick Start

```bash
npm install @importcsv/react
```

```tsx
import { CSVImporter } from '@importcsv/react';

<CSVImporter
  columns={[
    { id: 'name', label: 'Name', validators: [{ type: 'required' }] },
    { id: 'email', label: 'Email', type: 'email' }
  ]}
  onComplete={data => console.log(data)}
/>
```
